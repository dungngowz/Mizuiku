var idItem = $("#DisplayLoadControl_ctl00_ctl00_hdIid").val();
var content = $("#tbComment").val();
if (content === "") {
$("#tbComment").focus();
return;
}
loading(true);
jQuery.ajax({
url: weburl + "cms/display/KhoaHoc/Ajax/Ajax.aspx",
type: "POST",
dataType: "json",
data: {
"action": "comment",
"iidCmt": iid,
"idItem": idItem,
"content": content
},
success: function (res) {
loading(false);
$("#tbComment").val('');
$("#success .tac").html("Bình luận của bạn sẽ được kiểm duyệt bởi quản trị của Website trước khi được cập nhật. Xin cảm ơn.");
$(".success_open").click();
},
error: function (error) {//Lỗi xảy ra
loading(false);
alert("Hệ thống đang bận, bạn vui lòng thử lại sau!");
}
});
}
